```javascript
function isGpsRequired() {
  const cache = CacheService.getScriptCache();
  let status = cache.get("GPS_RUNTIME_STATUS");
  if (status === null) {
    status = PropertiesService.getScriptProperties().getProperty("GPS_ENABLED") || "true";
    cache.put("GPS_RUNTIME_STATUS", status, 600); // 10 นาที
  }
  return status === "true";
}

function updateGpsStatus(payload) {
  const actor = verifyAdminSession_(payload.auth); 
  if (actor.role !== "admin") throw new Error("Unauthorized");
  
  const newStatus = !!payload.enabled;
  const props = PropertiesService.getScriptProperties();
  props.setProperty("GPS_ENABLED", String(newStatus));
  CacheService.getScriptCache().remove("GPS_RUNTIME_STATUS");

  logAudit_("GPS_TOGGLE", {
    actor: actor.email,
    action: newStatus ? "ENABLED" : "DISABLED",
    reason: payload.reason,
    adminLat: payload.lat, // พิกัดยืนยันตัวตน
    adminLng: payload.lng,
    timestamp: new Date().toISOString()
  });
}
```

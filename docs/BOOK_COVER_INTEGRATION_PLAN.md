export function resolveImageUrl(item) {
  if (item?.coverUrl && item.coverUrl.trim() !== "") return item.coverUrl;
  return '/assets/img/default-cover.svg';

### 3.2 Backend: อัปโหลดไฟล์ (Module_Media.gs)
```javascript
function uploadBookCover(payload) {
  const actor = assertManageStaff_(payload.auth);
  // อ้างอิงโฟลเดอร์จาก Config
  const folder = DriveApp.getFolderById(CONFIG.BOOKS_COVER_FOLDER_ID);
  const file = folder.createFile(payload.blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return { url: file.getUrl(), id: file.getId() };
}
```
